% ─── Inclusión de figuras externas con fallback ───────────────────────────────
\newcommand{\incluirfigura}[4]{%
  \begin{figure}[H]
    \caption{#3}
    \label{#4}
    \centering
    \IfFileExists{#1}{%
      \includegraphics[width=#2]{#1}%
    }{%
      \fbox{\parbox[c][2.2in][c]{#2}{\centering\small Coloque la imagen en la carpeta \texttt{figuras/}\\#1}}%
    }
  \end{figure}%
}

% ─── Comando de figura APA 7 ──────────────────────────────────────────────────
\newcommand{\apafigure}[6]{%
  \begin{figure}[H]
    \refstepcounter{figure}\label{#1}%
    \noindent\textbf{Figura #2}\\[2pt]%
    \noindent\textit{#3}\\[8pt]%
    \centering
    \IfFileExists{#5}{%
      \includegraphics[width=#4]{#5}%
    }{%
      \fbox{\parbox[c][2.2in][c]{#4}{\centering Imagen no disponible:\\#5}}%
    }%
    \\[8pt]%
    \begin{minipage}{#4}%
      \small\textit{Nota.} #6
    \end{minipage}
  \end{figure}%
}
